# Vercel 배포 시 필요한 환경 변수들
# Vercel 대시보드 → Settings → Environment Variables에서 추가하세요

# Firebase 설정 (실제 키로 교체 필요)
NEXT_PUBLIC_FIREBASE_API_KEY=your_firebase_api_key
NEXT_PUBLIC_FIREBASE_AUTH_DOMAIN=your_project.firebaseapp.com
NEXT_PUBLIC_FIREBASE_PROJECT_ID=your_project_id
NEXT_PUBLIC_FIREBASE_STORAGE_BUCKET=your_project.appspot.com
NEXT_PUBLIC_FIREBASE_MESSAGING_SENDER_ID=your_sender_id
NEXT_PUBLIC_FIREBASE_APP_ID=your_app_id

# Stripe 설정 (실제 키로 교체 필요)
STRIPE_SECRET_KEY=sk_test_your_stripe_key
STRIPE_PUBLISHABLE_KEY=pk_test_your_stripe_key
STRIPE_WEBHOOK_SECRET=whsec_your_webhook_secret

# Kakao 설정 (실제 키로 교체 필요)
NEXT_PUBLIC_KAKAO_APP_KEY=your_kakao_app_key
KAKAO_ADMIN_KEY=your_kakao_admin_key
NEXT_PUBLIC_KAKAO_CHANNEL_ID=your_channel_id

# 기타 설정
NEXT_PUBLIC_BASE_URL=https://your-vercel-domain.vercel.app

# 배포 후 실제 도메인으로 변경
